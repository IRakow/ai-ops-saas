Setup Error Detection

Drop this snippet into your app to send errors and events to AI Ops. When something breaks, the pipeline triggers automatically.

How it works: The snippet POSTs error events to your intake endpoint. AI Ops receives them, classifies the issue, and dispatches the agent pipeline to investigate and open a fix. 
<script>
(function() {
  var AI_OPS_KEY    = "{{ api_key }}";
  var AI_OPS_INTAKE = "{{ intake_url }}";

  function send(payload) {
    try {
      navigator.sendBeacon(AI_OPS_INTAKE, JSON.stringify(payload));
    } catch (e) {
      var xhr = new XMLHttpRequest();
      xhr.open("POST", AI_OPS_INTAKE, true);
      xhr.setRequestHeader("Content-Type", "application/json");
      xhr.setRequestHeader("X-AI-Ops-Key", AI_OPS_KEY);
      xhr.send(JSON.stringify(payload));
    }
  }

  window.addEventListener("error", function(e) {
    send({
      type: "uncaught_error",
      message: e.message,
      source: e.filename,
      lineno: e.lineno,
      colno: e.colno,
      stack: e.error ? e.error.stack : null,
      url: window.location.href,
      ts: Date.now()
    });
  });

  window.addEventListener("unhandledrejection", function(e) {
    send({
      type: "unhandled_rejection",
      message: String(e.reason),
      stack: e.reason && e.reason.stack ? e.reason.stack : null,
      url: window.location.href,
      ts: Date.now()
    });
  });

  window._aiOps = { capture: send, key: AI_OPS_KEY };
})();
</script>

Paste this before </body> in your HTML, or load it via your asset pipeline. It auto-captures uncaught JS errors and unhandled promise rejections with no extra configuration.

Send POST requests here with X-AI-Ops-Key: {{ api_key }} in the header to submit events from any environment.

Keep this private. Rotate it any time from your dashboard at {{ saas_domain }}.

Not ready to add the snippet yet? You can skip this step and add it later from your dashboard. The pipeline also accepts events from server-side integrations, webhooks, and the REST API.
